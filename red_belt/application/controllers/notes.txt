SELECT users.first_name as first, users.last_name as last, users2.first_name as friend_first, users2.last_name as friend_last
FROM users
LEFT JOIN friendships ON friendships.user_id = users.id
LEFT JOIN users AS users2 ON users2.id = friendships.friend_id
ORDER BY users2.last_name;
